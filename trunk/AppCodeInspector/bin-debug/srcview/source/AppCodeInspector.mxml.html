        import mx.utils.ColorUtil;

        public function onMouseUp():void
        {
            var selection:TextRange = txtRitchText.selection;
            var selLength:int = selection.endIndex - selection.beginIndex;
            if((selLength)) 
            {
                var str:String = txtRitchText.selection.text;
                txtDebug.htmlText = "<b>Texto Selecionado:</b>\n" + str;
                if(selection.color != StyleManager.getColorName("red"))
                {
                    selection.color = "red";
                    txtDebug.htmlText += "\nSinalizou a seleção como <font color='#FF0000'><b>INCORRETO</b></font>"; 
                } 
                else
                {
                    selection.color = "0b333c";
                    txtDebug.htmlText += "\nSinalizou a seleção como <b>CORRETO</b>";
                }
            }
        }
